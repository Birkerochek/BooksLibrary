//Book.php

getTitle() - получить название книги
getAuthor() - получить автора книги
getYear() - получить год издания
getAvailability() - проверить доступность книги
setAvailability() - установить статус доступна ли книга

//Reader.php
getName() - получить имя читателя
getEmail() - получить email читателя

borrowBooks() - если книга доступа ( getAvailability() ), в массив взятых
книг добавляется эта книга и её доступность для других становится на false

returnBook() - возвращаем ключ найденного элемента в массиве, удаляем найденный
элемент из массива borroweBooks, меняем состояние доступности книги (setAvailability())

//Library.php

addBook() - добавить книгу
addReader() - добавить читателя

findBook() - ищем книгу с помощью перебора массива books и получаем title каждой
книги

listBooks() - получаем список книг
